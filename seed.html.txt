<html>
<body style="background:#080808;color:#fff;font-family:sans-serif;padding:40px;">
<h2>Seeding VibeReel Database...</h2>
<p id="status">Starting...</p>
<script type="module">
import { createClient } from 'https://cdn.jsdelivr.net/npm/@supabase/supabase-js@2/+esm';
const supabase = createClient('__SUPABASE_URL__', '__SUPABASE_KEY__');
const TMDB_KEY = '__TMDB_API_KEY__';

async function seed() {
    const genres = [28, 35, 18, 27, 10749, 53, 12, 878];
    let allMovies = [];
    
    for (const genre of genres) {
        const res = await fetch(`https://api.themoviedb.org/3/discover/movie?api_key=${TMDB_KEY}&with_genres=${genre}&sort_by=popularity.desc&page=1`);
        const data = await res.json();
        allMovies = allMovies.concat(data.results.slice(0, 5));
    }
    
    const unique = [...new Map(allMovies.map(m => [m.id, m])).values()];
    
    const genreMap = {28:"Action",12:"Adventure",16:"Animation",35:"Comedy",80:"Crime",99:"Documentary",18:"Drama",10751:"Family",14:"Fantasy",36:"History",27:"Horror",10402:"Music",9648:"Mystery",10749:"Romance",878:"Sci-Fi",53:"Thriller",10752:"War",37:"Western"};
    
    const movies = unique.map(m => ({
        tmdb_id: m.id,
        title: m.title,
        overview: m.overview,
        poster_path: m.poster_path,
        backdrop_path: m.backdrop_path,
        release_date: m.release_date,
        runtime: 120,
        vote_average: m.vote_average,
        vote_count: m.vote_count,
        genres: m.genre_ids.map(id => genreMap[id]).filter(Boolean),
        platforms: ["Netflix","Prime"],
        moods: [],
        occasions: [],
        categories: [],
        actors: [],
        director: "TBD"
    }));
    
    const { error } = await supabase.from('movies').upsert(movies, { onConflict: 'tmdb_id' });
    
    document.getElementById('status').textContent = error 
        ? 'Error: ' + error.message 
        : `Success! Added ${movies.length} movies. You can delete this file now.`;
}
seed();
</script>
